9.28 JS学习笔记

一、this的作用域判别

var o = {
    m:function () {
    var self = this;
    console.log(this === o);  //输出为true
    f();
    function f() {
        console.log(this === o);//该处this的值为全局对象或undefined，输出false
        console.log(this.a);
        console.log(self === o); //输出为true
        }
    },
    a:1
};
o.m();

this在改对象的属性嵌套函数中，子函数的this所指与父函数的this不同，为全局对象或undefined,而父函数则为该对象。

二.参数列表的使用

function max(/*...*/){
    var max = Number.NEGATIVE_INFINITY;
    for(var i = 0;i<arguments.length;i++){
        if(max < arguments[i]) max = arguments[i];
       }
    return max;
}

var a = max(1,3,5,7,9,555,10002,654);
console.log(a);   //操作参数列表，使用任意数目参数

参数列表作为一个数组，可以使得函数可以接受超出给出参数数量的参数进行运算。

三.函数的属性

uniqueInteger.counter = 0;
function uniqueInteger(){
    return uniqueInteger.counter++;
}

console.log(uniqueInteger()); //输出0
console.log(uniqueInteger()); //输出1

函数可以设定一个值作为属性（？），且属性值并不会不变，可以进行存储、保留

四.函数作用域声明

var scope = "global scope";
function checkscope(){
    var scope = "local scope";
    function f(){
        return scope;
       }
return f();
}

console.log(checkscope());  //输出localscope
console.log(scope);            //输出globalscope

//另一种情况：
scope = "global scope";
function checkscope(){
    scope = "local scope";
    function f(){
        return scope;
    }
    return f();
}
console.log(checkscope());   //输出localscope
console.log(scope);          //输出localscope
//局部变量声明对全局变量进行了修改

如果在不加var或其他对变量声明的话，局部变量的声明可以对全局变量进行修改，覆盖。

五.bind函数

//1.将函数绑定在对象中
function f(y){return this.x+y;}
var o = {x:1};
var g = f.bind(o);

//2.绑定实际参数
var sum = function(x,y){return x+y;}
var succ = sum.bind(null,1);
//第一个参数null是绑定的对象，第二个则为x

六.不完全函数

function array(a,n){return Array.prototype.slice.call(a,n||0);}
function test(f){
    var args = arguments;
    for(var i in args) console.log(args[i]); //输出[Function: f]、2
    return function(){
        var a = array(args,1);  //从索引为1开始切片，包含索引为1
        for(var i in a) console.log(a[i]);   //输出2  
        a = a.concat(array(arguments));
        for(var i in a) console.log(a[i]);   //输出2、3、4
        return f.apply(this,a);
    }
}

var f = function(x,y,z){return x*(y-z);};
test(f,2)(3,4);

后续括号中参数作为子函数的参数传入且可以使用arguments数组进行调用。
